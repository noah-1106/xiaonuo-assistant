// authService.js - жү©еұ•йӘҢиҜҒз Ғз®ЎзҗҶ
const verificationCodes = new Map();

class AuthService {
  // з”ҹжҲҗйӘҢиҜҒз Ғ
  generateCode() {
    return Math.floor(100000 + Math.random() * 900000).toString();
  }

  // еӯҳеӮЁйӘҢиҜҒз Ғ
  storeCode(key, code, expireMinutes = 5) {
    verificationCodes.set(key, {
      code,
      expireAt: Date.now() + expireMinutes * 60 * 1000
    });
  }

  // йӘҢиҜҒйӘҢиҜҒз Ғ
  verifyCode(key, code) {
    const storedCode = verificationCodes.get(key);
    if (!storedCode) {
      return false;
    }

    if (storedCode.code !== code) {
      return false;
    }

    if (Date.now() > storedCode.expireAt) {
      verificationCodes.delete(key);
      return false;
    }

    // жё…йҷӨе·ІдҪҝз”Ёзҡ„йӘҢиҜҒз Ғ
    verificationCodes.delete(key);
    return true;
  }

  // жё…зҗҶиҝҮжңҹйӘҢиҜҒз Ғ
  cleanupExpiredCodes() {
    const now = Date.now();
    for (const [key, value] of verificationCodes.entries()) {
      if (now > value.expireAt) {
        verificationCodes.delete(key);
      }
    }
  }
}

module.exports = new AuthService();
